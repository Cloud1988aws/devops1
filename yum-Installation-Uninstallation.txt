Installation (root)
======================
 sudo rm -f /etc/yum.repos.d/pgdg*.repo
 sudo rm -f /etc/yum.repos.d/postgresql*.repo
 sudo yum clean all
 sudo rm -rf /var/cache/yum
 yum repolist
 sudo rpm -e pgdg-redhat-repo
 yum install -y https://download.postgresql.org/pub/repos/yum/reporpms/EL-7-x86_64/pgdg-redhat-repo-latest.noarch.rpm
 ls -al
 less pgdg-redhat-all.repo 
 sudo yum install -y postgresql15-server
  yum --disablerepo=pgdg12
 vi pgdg-redhat-all.repo 
 sudo yum install -y postgresql15-server
 sudo yum install -y postgresql15-contrib.x86_64
 rpm -qa |grep postgres
 sudo /usr/pgsql-15/bin/postgresql-15-setup initdb
 ls -al /var/lib/pgsql/15/
 ls -al /var/lib/pgsql/15/data/
 history
 systemctl list-unit-file
 systemctl list-unit-files
 systemctl list-unit-files|grep postgr
 systemctl status postgresql-15.service 
 systemctl enable  postgresql-15.service 
 systemctl start  postgresql-15.service 
 ps -ef|grep -i postgres
 systemctl stop  postgresql-15.service 
 ps -ef|grep -i postgres


UnInstallation (root)
===================
systemctl stop postgresql-15.service 
ps -ef|grep -i postgres
yum remove postgresql15-server.x86_64 
rm -rf /var/lib/pgsql/15/
sudo yum remove postgresql15 postgresql15-libs
